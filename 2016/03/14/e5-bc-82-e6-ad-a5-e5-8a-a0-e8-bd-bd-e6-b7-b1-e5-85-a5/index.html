异步加载深入

提高用户体验—-使用缓存

  • Lru算法

    • Lru：Least Recently Used 近期最少使用算法
    • Android提供了LruCache类来实现这个算法

  • 以内存换率
    以下奉上在慕课网中学习到的线程加载图片和异步任务加载图片加上缓存机制的代码：
    `
    /**

    • Created by addi on 2016/3/13 0013.
      */
      public class ImageLoader {

    private ImageView tampView; //暂存iamgeView
    private String mURL;
    private LruCache<String,Bitmap> mCaches;//缓存

    /**

    • 构造函数
      */
      public ImageLoader(){
      int maxSize = (int) Runtime.getRuntime().maxMemory();
      int cacheSize = maxSize/4;
      //初始化Lru缓存
      mCaches = new LruCache<String,Bitmap>(cacheSize){
      @Override
protected int sizeOf(String key, Bitmap value) {
    return value.getByteCount();
}
      };
      }

    /**

    • 线程方式加载图片时用于更新UI的Handler
      */
      private Handler handler = new Handler(){
      @Override
      public void handleMessage(Message msg) {
      super.handleMessage(msg);
if(tampView.getTag().equals(mURL))
tampView.setImageBitmap((Bitmap) msg.obj);
      }
      };

    /**

    • 线程方式
      */
      public void getImage(ImageView image, final String url){
      tampView = image;
      mURL = url;
      new Thread(){
      @Override
public void run() {
    super.run();
    Bitmap bitmap = getBitmapFromURL(url);
    Message message =Message.obtain();
    message.obj = bitmap;
    handler.sendMessage(message);
}
      }.start();
      }

    /**

    • 从网络中获取图片
    • @param url
    • @return
    • @throws IOException
      */
      public Bitmap getBitmapFromURL(String url) {
      Bitmap bitmap;
      InputStream is = null;
      try {
      HttpURLConnection conn = (HttpURLConnection) new URL(url).openConnection();
is = new BufferedInputStream(conn.getInputStream());
bitmap = BitmapFactory.decodeStream(is);
conn.disconnect();
return bitmap;
      } catch (IOException e) {
      e.printStackTrace();
      }finally{
      try {
    is.close();
} catch (IOException e) {
    e.printStackTrace();
}
      }
      return null;
      }

    /**

    • 将图片添加到缓存
    • 注意：要先判读缓存中市是否已经存在这个图片
    • @param key
    • @param bitmap
      */
      private void addBitmapToCache(String key,Bitmap bitmap){
      if(mCaches.get(key) == null){
      mCaches.put(key,bitmap);
      }
      }

    /**

    • 根据key值从缓存中获取图片
    • @param key
    • @return
      */
      private Bitmap getBitmapFromCache(String key){
      return mCaches.get(key);
      }

    /**

    • 异步任务方式加载图片
    • @param imageView
    • @param url
      */

    public void showImageByAsyncTask(ImageView imageView,String url){

    Bitmap bitmap =  getBitmapFromCache(url);
if(bitmap==null) {
    new ImageAsyncTask(imageView, url).execute(url); //缓存中没有图片，启动异步任务获取
}else {
    imageView.setImageBitmap(bitmap);       //缓存中存在直接设置图片
}

    }

    /**

    • 异步任务类
      */
      //后：：添加缓存机制到异步任务中
      class ImageAsyncTask extends AsyncTask<String,Void,Bitmap>{

      private ImageView imageView;
      private String url;

      public ImageAsyncTask(ImageView imageView, String url) {
      this.imageView = imageView;
      this.url = url;
      }

      @Override
      protected Bitmap doInBackground(String… params) {
      //从网络中获取图片
      Bitmap bitmap = getBitmapFromURL(params[0]);
      String url = params[0];
      if(bitmap!=null){

      addBitmapToCache(url,bitmap); //将图片加至缓存中

      }
      return bitmap;
      }

      @Override
      protected void onPostExecute(Bitmap bitmap) {
      super.onPostExecute(bitmap);
      if(imageView.getTag().equals(url))
      imageView.setImageBitmap(bitmap);
      }
      }

    }

`
